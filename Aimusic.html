# Cell 4: Frequency + Rhythm Entropy

def get_rhythmic_entropy(signal, sr):
    onset_env = librosa.onset.onset_strength(y=signal, sr=sr)
    norm_env = onset_env / np.sum(onset_env)
    beat_entropy = entropy(norm_env)
    return beat_entropy, onset_env

rh_entropy, onset_env = get_rhythmic_entropy(y, sr)
print(f"Rhythmic Entropy: {rh_entropy:.4f}")

# ------------------------------
# Cell 6: Define Simple Music-Contradiction Gym Env
# ------------------------------
class MusicContradictionEnv(gym.Env):
    def __init__(self):
        super(MusicContradictionEnv, self).__init__()
        self.action_space = spaces.Discrete(3)  # 0=Harmony, 1=Polyrhythm, 2=Dissonance
        self.observation_space = spaces.Box(low=0, high=1, shape=(1,), dtype=np.float32)
        self.state = np.array([0.5])  # Start at neutral entropy
        self.entropy_history = []

    def step(self, action):
        if action == 0:  # Harmony
            delta = -0.05
        elif action == 1:  # Polyrhythm (nested contradiction)
            delta = 0.02
        else:  # Dissonance
            delta = 0.05

        self.state += delta + np.random.normal(0, 0.01)
        self.state = np.clip(self.state, 0, 1)
        self.entropy_history.append(self.state[0])

        # Reward = low entropy, but small entropy spikes (resolution) are OK
        reward = 1.0 - abs(self.state[0] - 0.3)
        done = len(self.entropy_history) >= 50
        return self.state, reward, done, {}

    def reset(self):
        self.state = np.array([0.5])
        self.entropy_history = []
        return self.state

# ------------------------------
# Cell 7: Train Random Agent
# ------------------------------
env = MusicContradictionEnv()
episodes = 10
all_rewards = []

for e in range(episodes):
    state = env.reset()
    total_reward = 0
    for step in range(50):
        action = env.action_space.sample()  # Random for now
        state, reward, done, _ = env.step(action)
        total_reward += reward
        if done:
            break
    all_rewards.append(total_reward)

print(f"Average Reward: {np.mean(all_rewards):.2f}")
